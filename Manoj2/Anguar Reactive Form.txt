import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Observable } from 'rxjs';
import { Product } from './Product';
@Injectable({
  providedIn: 'root'
})
export class MyserviceService {
  URL: string = "http://localhost:3000/products";
  
  constructor(private h: HttpClient) { }


  getProducts(): Observable<any>{
    return this.h.get<any>(this.URL);    
  }
  addProduct(p: Product): Observable<any>{
    return this.h.post(this.URL, p, { responseType: 'json'});
  }
  modifyProduct(id:string,p: Product): Observable<any>{
    return this.h.put(this.URL+"/"+id, p, { responseType: 'json' });
  }
  delProduct(id: string): Observable<any>{
    return this.h.delete(this.URL + "/" + id,{responseType:'json'});
  }
  getProduct(id: string): Observable<any>{
    return this.h.get<any>(this.URL + "/" + id);
  }


}
