Definer vs Invoker Rights (AUTHID CURRENT_USER)

--    Definer = the owner (creator) of the PL/SQL object
When we say “Definer”, we mean:

The database user who CREATES the procedure, function, package, or view

by default he has rights to access the subprograms

--   Invoker Rights (AUTHID CURRENT_USER)

What it means
Invoker rights means a PL/SQL program executes with the privileges of the user who calls it, 
not the user who owns it.

(HR User )

CREATE OR REPLACE PROCEDURE p1
AUTHID CURRENT_USER
AS
  v_sal hr.emp.sal%TYPE;
BEGIN
  SELECT sal INTO v_sal FROM hr.emp WHERE empid = 101;
  DBMS_OUTPUT.put_line(v_sal);
END;

   Package State

CREATE OR REPLACE PACKAGE counter_pkg AS
  total_calls NUMBER := 0;
  PROCEDURE increment;
END counter_pkg;


CREATE OR REPLACE PACKAGE BODY counter_pkg AS
  PROCEDURE increment IS
  BEGIN
    total_calls := total_calls + 1;
    DBMS_OUTPUT.PUT_LINE('Total Calls: ' || total_calls);
  END;
END counter_pkg;

output Total calls: = 2

it maintain the state between procedure calls


*    Package Initialization Block

Package State:

Package state is the persistent value of package variables maintained across calls within a session.

Package Initialization Block:

A block in the package body that executes once per session to initialize package variables.


create or replace package countPkg as
 tcount number;
procedure p1;
end;


CREATE OR REPLACE PACKAGE BODY countPkg AS

procedure p1 is
begin

dbms_output.put_line(' total count '||tcount);
end p1;

begin
tcount:=100;
dbms_output.put_line('Main Package count');

end countPkg;

begin
countPkg.p1;
end;

Private vs Public Subprograms


CREATE OR REPLACE PACKAGE math_pkg AS
  FUNCTION add_nums(a NUMBER, b NUMBER) RETURN NUMBER; -- PUBLIC
END math_pkg;
/
CREATE OR REPLACE PACKAGE BODY math_pkg AS

  FUNCTION add_nums(a NUMBER, b NUMBER) RETURN NUMBER IS
  BEGIN
    RETURN a + b;
  END add_nums;

  FUNCTION multiply_nums(a NUMBER, b NUMBER) RETURN NUMBER IS
  BEGIN
    RETURN a * b;
  END multiply_nums;

END math_pkg;
/
SELECT math_pkg.add_nums(2, 3) FROM dual;     -- Works
SELECT math_pkg.multiply_nums(2, 3) FROM dual; -- Error

Dynamic SQL
    What is Dynamic SQL?
	Dynamic SQL means building and executing SQL statements at runtime
 	instead of writing fixed (static) SQL at compile time.
    
  static SQL
          SELECT salary INTO v_sal FROM employees WHERE emp_id = 101;
        
        sql_stmt := 'SELECT salary FROM employees WHERE emp_id = :id';
        EXECUTE IMMEDIATE sql_stmt INTO v_sal USING 101;

    EXECUTE IMMEDIATE
     Dynamic DDL, DML

        DECLARE
  v_sql VARCHAR2(200);
BEGIN
  v_sql := 'INSERT INTO employees (emp_id, emp_name)
            VALUES (:1, :2)';

  EXECUTE IMMEDIATE v_sql USING 101, 'Manoj';
END;

  DECLARE
  v_sql VARCHAR2(200);
BEGIN
  v_sql := 'UPDATE employees SET salary = :sal WHERE emp_id = :id';

  EXECUTE IMMEDIATE v_sql USING 60000, 101;
END;

Concurrency & Locking

   What is Concurrency?
Concurrency means multiple users or sessions accessing and modifying database data at the same time.
Example:

User A updates a row in EMPLOYEES
User B tries to update the same row at the same time

    Lock types
   Oracle has automatic locking 
          when update product set price=100 where pid=100;
             this row is locked till its gets updated

    Deadlock avoidance
  What is a Deadlock?
A deadlock occurs when:

Session A holds Lock 1 and waits for Lock 2
Session B holds Lock 2 and waits for Lock 1
 Neither can proceed.


  eg
        Session1 
  UPDATE emp SET sal = sal + 100 WHERE empno = 7369;
-- waits
UPDATE dept SET loc = 'NY' WHERE deptno = 10;
  
   Session2
  UPDATE dept SET loc = 'NY' WHERE deptno = 10;
-- waits
UPDATE emp SET sal = sal + 100 WHERE empno = 7369;

Oracle automatically detects deadlock and it rollback one session 




--     Autonomous transactions
  
 What is an Autonomous Transaction?
An autonomous transaction is a separate, independent transaction inside another transaction.

It has its own commit/rollback
Does not affect the parent transaction

  --      DBMS_SQL

     DBMS_SQL is used to execute dynamic SQL statements where the SQL structure 
    (number of columns, data types, etc.) is not known at compile time.

     DECLARE
  cur NUMBER;
  rows NUMBER;
BEGIN
  cur := DBMS_SQL.OPEN_CURSOR;
  DBMS_SQL.PARSE(cur, 'UPDATE emp SET sal = sal + 500 WHERE deptno = 10',
                 DBMS_SQL.NATIVE);
  rows := DBMS_SQL.EXECUTE(cur);
  DBMS_SQL.CLOSE_CURSOR(cur);
END;
